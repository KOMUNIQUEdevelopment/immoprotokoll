philipp@cloudshell:~ (immoprotokoll-492520)$ # 1. Statische IP reservieren
gcloud compute addresses create www-redirect-ip --global

# 2. URL-Map mit 301-Redirect erstellen
cat > /tmp/redirect-map.yaml << 'EOF'
defaultUrlRedirect:
  httpsRedirect: true
  redirectResponseCode: MOVED_PERMANENTLY_DEFAULT
  hostRedirect: immoprotokoll.com
  prefixRedirect: /
  pathRedirect: /
  stripQuery: false
EOF
gcloud compute url-maps import www-redirect-map \
  --source /tmp/redirect-map.yaml --global

# 3. SSL-Zertifikat für www anlegen
gcloud compute ssl-certificates create www-cert \
  --domains=www.immoprotokoll.com --global

# 4. HTTPS-Proxy erstellen
gcloud compute target-https-proxies create www-https-proxy \
  --url-map=www-redirect-map \
  --ssl-certificates=www-cert --global

# 5. Forwarding Rule (IP → Proxy)
gcloud compute forwarding-rules create www-https-rule \
  --address=www-redirect-ip \
  --target-https-proxy=www-https-proxy \
  --ports=443 --global

# IP anzeigen → für DNS-Record
gcloud compute addresses describe www-redirect-ip --global --format="value(address)"
